30  package com.jcabi.aspects.version;
31  
32  import org.hamcrest.MatcherAssert;
33  import org.hamcrest.Matchers;
34  import org.junit.jupiter.api.Test;
35  
36  /**
37   * Unit tests for {@link Version}.
38   * @since 0.23
39   */
40  public final class VersionTest {
41  
42      /**
43       * Version.CURRENT can contain actual project version and not a
44       * "${project.version}" placeholder.
45       */
46      @Test
47      public void containsCorrectVersionNumber() {
48          MatcherAssert.assertThat(
49              Version.CURRENT.projectVersion(),
50              Matchers.not(
51                  Matchers.equalTo("${projectVersion}")
52              )
53          );
54      }
55  
56      /**
57       * Version.CURRENT can contain actual build number and not a
58       * "${buildNumber}" placeholder.
59       */
60      @Test
61      public void containsCorrectBuildNumber() {
62          MatcherAssert.assertThat(
63              Version.CURRENT.buildNumber(),
64              Matchers.not(
65                  Matchers.equalTo("${buildNumber}")
66              )
67          );
68      }
69  }
