30  package com.jcabi.aspects;
31  
32  import java.util.concurrent.TimeUnit;
33  import java.util.concurrent.atomic.AtomicInteger;
34  import org.hamcrest.MatcherAssert;
35  import org.hamcrest.Matchers;
36  import org.junit.jupiter.api.Test;
37  
38  /**
39   * Test case for {@link RetryOnFailure} annotation and its implementation.
40   * @since 0.0.0
41   */
42  @SuppressWarnings("PMD.DoNotUseThreads")
43  public final class RetryOnFailureTest {
44  
45      /**
46       * RetryOnFailure can force duplicate execution of the same method.
47       */
48      @Test
49      public void executesMethodManyTimes() {
50          final AtomicInteger count = new AtomicInteger();
51          new Runnable() {
52              @Override
53              @RetryOnFailure(verbose = false, unit = TimeUnit.SECONDS, delay = 1)
54              public void run() {
55                  if (count.incrementAndGet() < 2) {
56                      throw new IllegalArgumentException(
57                          "this exception should be caught and swallowed"
58                      );
59                  }
60              }
61          } .run();
62          MatcherAssert.assertThat(count.get(), Matchers.greaterThan(0));
63      }
64  
65      /**
66       * RetryOnFailure can retry when Error types are thrown.
67       */
68      @Test
69      public void retriesOnError() {
70          final AtomicInteger count = new AtomicInteger();
71          new Runnable() {
72              @Override
73              @RetryOnFailure(verbose = false, unit = TimeUnit.SECONDS, delay = 1)
74              public void run() {
75                  if (count.incrementAndGet() < 2) {
76                      throw new AssertionError("Should be caught and ignored.");
77                  }
78              }
79          } .run();
80          MatcherAssert.assertThat(count.get(), Matchers.greaterThan(0));
81      }
82  
83  }
