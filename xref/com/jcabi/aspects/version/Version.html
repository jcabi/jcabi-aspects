30  package com.jcabi.aspects.version;
31  
32  /**
33   * Current version of the project. Generated from a template at build time.
34   * @since 0.23
35   */
36  public enum Version {
37      /**
38       * Current version.
39       */
40      CURRENT("0.24.1", "4a53c0f");
41  
42      /**
43       * Project version.
44       */
45      private final String version;
46  
47      /**
48       * Build number.
49       */
50      private final String build;
51  
52      /**
53       * Public ctor.
54       * @param ver Maven's project.version property
55       * @param buildnum Maven's buildNumber property created with
56       *  buildnumber-maven-plugin
57       */
58      Version(final String ver, final String buildnum) {
59          this.version = ver;
60          this.build = buildnum;
61      }
62  
63      /**
64       * Returns project version number.
65       * @return Project version number
66       */
67      public String projectVersion() {
68          return this.version;
69      }
70  
71      /**
72       * Returns project build number.
73       * @return Build number
74       */
75      public String buildNumber() {
76          return this.build;
77      }
78  }
