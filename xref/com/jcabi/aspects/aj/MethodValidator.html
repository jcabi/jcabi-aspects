30  package com.jcabi.aspects.aj;
31  
32  import java.lang.reflect.Constructor;
33  import java.lang.reflect.Method;
34  import java.util.Set;
35  import javax.validation.ConstraintViolation;
36  import javax.validation.ConstraintViolationException;
37  import javax.validation.Validation;
38  import javax.validation.Validator;
39  import org.aspectj.lang.JoinPoint;
40  import org.aspectj.lang.annotation.AfterReturning;
41  import org.aspectj.lang.annotation.Aspect;
42  import org.aspectj.lang.annotation.Before;
43  import org.aspectj.lang.reflect.ConstructorSignature;
44  import org.aspectj.lang.reflect.MethodSignature;
45  
46  /**
47   * Validates method calls.
48   *
49   * <p>We do this manual processing of {@code javax.validation.constraints.*}
50   * annotations only because
51   * JSR-303 in its current version doesn't support method level validation
52   * (see its Appendix C). At the moment we don't support anything expect these
53   * two annotations. We think that it's better to wait for JSR-303.
54   *
55   * <p>The class is thread-safe.
56   *
57   * @since 0.1.10
58   * @see <a href="http://beanvalidation.org/1.0/spec/#appendix-methodlevelvalidation">Appendix C</a>
59   * @see <a href="http://aspects.jcabi.com/jsr-303.html">How it works</a>
60   */
61  @Aspect
62  @SuppressWarnings({ "PMD.CyclomaticComplexity", "PMD.TooManyMethods" })
63  public final class MethodValidator {
64  
65      /**
66       * JSR-303 Validator.
67       */
68      private final transient Validator validator = Validation
69          .buildDefaultValidatorFactory()
70          .getValidator();
71  
72      /**
73       * Validate arguments of a method.
74       *
75       * <p>Try NOT to change the signature of this method, in order to keep
76       * it backward compatible.
77       *
78       * @param point Join point
79       * @checkstyle LineLength (3 lines)
80       */
81      @Before("execution(* *(.., @(javax.validation.* || javax.validation.constraints.*) (*), ..))")
82      public void beforeMethod(final JoinPoint point) {
83          if (this.validator != null) {
84              this.validateMethod(
85                  point.getThis(),
86                  ((MethodSignature) point.getSignature()).getMethod(),
87                  point.getArgs()
88              );
89          }
90      }
91  
92      /**
93       * Validate arguments of constructor.
94       *
95       * <p>Try NOT to change the signature of this method, in order to keep
96       * it backward compatible.
97       *
98       * @param point Join point
99       * @checkstyle LineLength (3 lines)
100      */
101     @Before("preinitialization(*.new(.., @(javax.validation.* || javax.validation.constraints.*) (*), ..))")
102     public void beforeCtor(final JoinPoint point) {
103         if (this.validator != null) {
104             @SuppressWarnings("unchecked")
105             final Constructor<Object> constructor = (Constructor<Object>)
106                 ((ConstructorSignature) point.getSignature())
107                     .getConstructor();
108             this.validateConstructor(
109                 constructor,
110                 point.getArgs()
111             );
112         }
113     }
114 
115     /**
116      * Validate method response.
117      *
118      * <p>Try NOT to change the signature of this method, in order to keep
119      * it backward compatible.
120      *
121      * @param point Join point
122      * @param result Result of the method
123      * @since 0.7.11
124      */
125     @AfterReturning
126         (
127             // @checkstyle LineLength (1 line)
128             pointcut = "execution(@(javax.validation.* || javax.validation.constraints.*) * *(..))",
129             returning = "result"
130         )
131     public void after(final JoinPoint point, final Object result) {
132         this.checkForViolations(
133             this.validator
134                 .forExecutables()
135                 .validateReturnValue(
136                     point.getThis(),
137                     new ImprovedJoinPoint(point).currentMethod(),
138                     result
139                 )
140         );
141     }
142 
143     /**
144      * Validates method parameters.
145      * @param object Object at pointcut
146      * @param method Method at pointcut
147      * @param args Parameters of the method
148      */
149     private void validateMethod(final Object object, final Method method,
150         final Object... args) {
151         this.checkForViolations(
152             this.validator
153                 .forExecutables()
154                 .validateParameters(
155                     object,
156                     method,
157                     args
158                 )
159         );
160     }
161 
162     /**
163      * Validates constructor parameters.
164      * @param ctr Constructor at pointcut
165      * @param args Parameters of the method
166      */
167     private void validateConstructor(final Constructor<Object> ctr,
168         final Object... args) {
169         this.checkForViolations(
170             this.validator
171                 .forExecutables()
172                 .validateConstructorParameters(
173                     ctr,
174                     args
175                 )
176         );
177     }
178 
179     /**
180      * Checks if violations set is empty and throws
181      * {@link ConstraintViolationException} if it isn't.
182      *
183      * @param violations JSR303 violations.
184      */
185     private void checkForViolations(
186         final Set<ConstraintViolation<Object>> violations) {
187         if (!violations.isEmpty()) {
188             throw new ConstraintViolationException(
189                 MethodValidator.pack(violations),
190                 violations
191             );
192         }
193     }
194 
195     /**
196      * Pack violations into string.
197      * @param errs All violations
198      * @return The full text
199      */
200     private static String pack(
201         final Iterable<ConstraintViolation<Object>> errs) {
202         final StringBuilder text = new StringBuilder(0);
203         for (final ConstraintViolation<?> violation : errs) {
204             if (text.length() > 0) {
205                 text.append("; ");
206             }
207             text.append(violation.getMessage());
208         }
209         return text.toString();
210     }
211 }
