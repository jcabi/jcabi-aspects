57  /**
58   * Validates method calls.
59   *
60   * <p>We do this manual processing of {@code javax.validation.constraints.*}
61   * annotations only because
62   * JSR-303 in its current version doesn't support method level validation
63   * (see its Appendix C). At the moment we don't support anything expect these
64   * two annotations. We think that it's better to wait for JSR-303.
65   *
66   * <p>The class is thread-safe.
67   *
68   * @author Yegor Bugayenko (yegor@tpc2.com)
69   * @version $Id$
70   * @since 0.1.10
71   * @see <a href="http://beanvalidation.org/1.0/spec/#appendix-methodlevelvalidation">Appendix C</a>
72   * @see <a href="http://aspects.jcabi.com/jsr-303.html">How it works</a>
73   */
74  @Aspect
75  @SuppressWarnings({ "PMD.CyclomaticComplexity", "PMD.TooManyMethods" })
76  public final class MethodValidator {
77  
78      /**
79       * JSR-303 Validator.
80       */
81      private final transient Validator validator = MethodValidator.build();
82  
83      /**
84       * Validate arguments of a method.
85       *
86       * <p>Try NOT to change the signature of this method, in order to keep
87       * it backward compatible.
88       *
89       * @param point Join point
90       * @checkstyle LineLength (3 lines)
91       */
92      @Before("execution(* *(.., @(javax.validation.* || javax.validation.constraints.*) (*), ..))")
93      public void beforeMethod(final JoinPoint point) {
94          if (this.validator != null) {
95              this.validate(
96                  point,
97                  MethodSignature.class.cast(point.getSignature())
98                      .getMethod()
99                      .getParameterAnnotations()
100             );
101         }
102     }
103 
104     /**
105      * Validate arguments of constructor.
106      *
107      * <p>Try NOT to change the signature of this method, in order to keep
108      * it backward compatible.
109      *
110      * @param point Join point
111      * @checkstyle LineLength (3 lines)
112      */
113     @Before("preinitialization(*.new(.., @(javax.validation.* || javax.validation.constraints.*) (*), ..))")
114     public void beforeCtor(final JoinPoint point) {
115         if (this.validator != null) {
116             this.validate(
117                 point,
118                 ConstructorSignature.class.cast(point.getSignature())
119                     .getConstructor()
120                     .getParameterAnnotations()
121             );
122         }
123     }
124 
125     /**
126      * Validate method response.
127      *
128      * <p>Try NOT to change the signature of this method, in order to keep
129      * it backward compatible.
130      *
131      * @param point Join point
132      * @param result Result of the method
133      * @checkstyle LineLength (4 lines)
134      * @since 0.7.11
135      */
136     @AfterReturning(
137         pointcut = "execution(@(javax.validation.* || javax.validation.constraints.*) * *(..))",
138         returning = "result"
139     )
140     public void after(final JoinPoint point, final Object result) {
141         final Method method = MethodSignature.class.cast(
142             point.getSignature()
143         ).getMethod();
144         if (method.isAnnotationPresent(NotNull.class) && result == null
145             && !method.getReturnType().equals(Void.TYPE)) {
146             throw new ConstraintViolationException(
147                 method.getAnnotation(NotNull.class).message(),
148                 new HashSet<ConstraintViolation<?>>(
149                     Collections.<ConstraintViolation<?>>singletonList(
150                         MethodValidator.violation(
151                             result,
152                             method.getAnnotation(NotNull.class).message()
153                         )
154                     )
155                 )
156             );
157         }
158         if (method.isAnnotationPresent(Valid.class) && result != null) {
159             final Set<ConstraintViolation<Object>> violations =
160                 this.validate(result);
161             if (!violations.isEmpty()) {
162                 throw new ConstraintViolationException(violations);
163             }
164         }
165     }
166 
167     /**
168      * Validate method at the given point.
169      * @param point Join point
170      * @param params Parameters (their annotations)
171      */
172     private void validate(final JoinPoint point, final Annotation[][] params) {
173         final Set<ConstraintViolation<?>> violations =
174             new HashSet<ConstraintViolation<?>>(0);
175         for (int pos = 0; pos < params.length; ++pos) {
176             violations.addAll(
177                 this.validate(pos, point.getArgs()[pos], params[pos])
178             );
179         }
180         if (!violations.isEmpty()) {
181             throw new ConstraintViolationException(
182                 MethodValidator.pack(violations),
183                 violations
184             );
185         }
186     }
187 
188     /**
189      * Validate one method argument against its annotations.
190      * @param pos Position of the argument in method signature
191      * @param arg The argument
192      * @param annotations Array of annotations
193      * @return A set of violations
194      * @todo #61 It's a temporary design, which enables only NotNull,
195      *  Valid, and Pattern annotations. In the future we should use
196      *  JSR-303 Validator, when they implement validation of values (see
197      *  their appendix C).
198      */
199     private Set<ConstraintViolation<?>> validate(final int pos,
200         final Object arg, final Annotation[] annotations) {
201         final Set<ConstraintViolation<?>> violations =
202             new HashSet<ConstraintViolation<?>>(0);
203         for (final Annotation antn : annotations) {
204             if (antn.annotationType().equals(NotNull.class)) {
205                 if (arg == null) {
206                     violations.add(
207                         MethodValidator.violation(
208                             String.format("param #%d", pos),
209                             NotNull.class.cast(antn).message()
210                         )
211                     );
212                 }
213             } else if (antn.annotationType().equals(Valid.class)
214                 && arg != null) {
215                 violations.addAll(this.validate(arg));
216             } else if (antn.annotationType().equals(Pattern.class)) {
217                 if (arg != null && !arg.toString()
218                     .matches(Pattern.class.cast(antn).regexp())) {
219                     violations.add(
220                         MethodValidator.violation(
221                             String.format("param #%d '%s'", pos, arg),
222                             Pattern.class.cast(antn).message()
223                         )
224                     );
225                 }
226             } else if (antn.annotationType()
227                 .isAnnotationPresent(Constraint.class)) {
228                 Logger.warn(
229                     this,
230                     "%[type]s annotation is not supported at the moment",
231                     antn.annotationType()
232                 );
233             }
234         }
235         return violations;
236     }

238     /**
239      * Create one simple violation.
240      * @param arg The argument passed
241      * @param msg Error message to show
242      * @return The violation
243      */
244     private static ConstraintViolation<?> violation(final Object arg,
245         final String msg) {
246         // @checkstyle AnonInnerLength (50 lines)
247         return new ConstraintViolation<String>() {
248             @Override
249             public String toString() {
250                 return String.format("%s %s", arg, msg);
251             }
252             @Override
253             public ConstraintDescriptor<?> getConstraintDescriptor() {
254                 return null;
255             }
256             @Override
257             public Object getInvalidValue() {
258                 return arg;
259             }
260             @Override
261             public Object getLeafBean() {
262                 return null;
263             }
264             @Override
265             public String getMessage() {
266                 return msg;
267             }
268             @Override
269             public String getMessageTemplate() {
270                 return msg;
271             }
272             @Override
273             public Path getPropertyPath() {
274                 return null;
275             }
276             @Override
277             public String getRootBean() {
278                 return "";
279             }
280             @Override
281             public Class<String> getRootBeanClass() {
282                 return String.class;
283             }
284             @Override
285             public Object[] getExecutableParameters() {
286                 return new Object[0];
287             }
288             @Override
289             public Object getExecutableReturnValue() {
290                 return null;
291             }
292             @Override
293             public <U> U unwrap(final Class<U> type) {
294                 return null;
295             }
296         };
297     }

299     /**
300      * Pack violations into string.
301      * @param errs All violations
302      * @return The full text
303      */
304     private static String pack(final Collection<ConstraintViolation<?>> errs) {
305         final StringBuilder text = new StringBuilder(0);
306         for (final ConstraintViolation<?> violation : errs) {
307             if (text.length() > 0) {
308                 text.append("; ");
309             }
310             text.append(violation.getMessage());
311         }
312         return text.toString();
313     }

315     /**
316      * Build validator.
317      * @return Validator to use in the singleton
318      */
319     @SuppressWarnings("PMD.AvoidCatchingThrowable")
320     private static Validator build() {
321         Validator val = null;
322         try {
323             val = Validation.buildDefaultValidatorFactory().getValidator();
324             Logger.info(
325                 MethodValidator.class,
326                 // @checkstyle LineLength (1 line)
327                 "JSR-303 validator %[type]s instantiated by jcabi-aspects 0.22/7e07127",
328                 val
329             );
330         } catch (final ValidationException ex) {
331             Logger.error(
332                 MethodValidator.class,
333                 // @checkstyle LineLength (1 line)
334                 "JSR-303 validator failed to initialize: %s (see http://aspects.jcabi.com/jsr-303.html)",
335                 ex.getMessage()
336             );
337         // @checkstyle IllegalCatch (1 line)
338         } catch (final Throwable ex) {
339             Logger.error(
340                 MethodValidator.class,
341                 "JSR-303 validator thrown during initialization: %[exception]s",
342                 ex
343             );
344         }
345         return val;
346     }
347 
348     /**
349      * Check validity of an object, when it is annotated with {@link Valid}.
350      * @param object The object to validate
351      * @return Found violations
352      * @param <T> Type of violations
353      */
354     @SuppressWarnings("PMD.AvoidCatchingThrowable")
355     private <T> Set<ConstraintViolation<T>> validate(final T object) {
356         Set<ConstraintViolation<T>> violations;
357         try {
358             violations = this.validator.validate(object);
359         // @checkstyle IllegalCatch (1 line)
360         } catch (final Throwable ex) {
361             Logger.error(
362                 this,
363                 // @checkstyle LineLength (1 line)
364                 "JSR-303 validator %[type]s thrown %s while validating %[type]s",
365                 this.validator,
366                 ex,
367                 object
368             );
369             violations = new HashSet<ConstraintViolation<T>>(0);
370         }
371         return violations;
372     }
