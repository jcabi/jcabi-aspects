30  package com.jcabi.aspects.aj;
31  
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.log.Logger;
34  import org.aspectj.lang.ProceedingJoinPoint;
35  import org.aspectj.lang.annotation.Around;
36  import org.aspectj.lang.annotation.Aspect;
37  import org.aspectj.lang.reflect.MethodSignature;
38  
39  /**
40   * Logs all exceptions thrown out of a method and swallow exception.
41   *
42   * @since 0.1.10
43   * @see com.jcabi.aspects.LogExceptions
44   * @checkstyle IllegalThrows (100 lines)
45   * @checkstyle NonStaticMethodCheck (100 lines)
46   */
47  @Aspect
48  @Immutable
49  public final class QuietExceptionsLogger {
50  
51      /**
52       * Catch exception and log it, the exception will be swallowed.
53       *
54       * <p>This aspect should be used only on void returning methods.
55       *
56       * <p>Try NOT to change the signature of this method, in order to keep
57       * it backward compatible.
58       *
59       * @param point Joint point
60       * @return The result of call
61       */
62      @Around
63          (
64              // @checkstyle StringLiteralsConcatenation (2 lines)
65              "execution(* * (..))"
66              + " && @annotation(com.jcabi.aspects.Quietly)"
67          )
68      @SuppressWarnings("PMD.AvoidCatchingThrowable")
69      public Object wrap(final ProceedingJoinPoint point) {
70          if (!((MethodSignature) point.getSignature()).getReturnType()
71              .equals(Void.TYPE)) {
72              throw new IllegalStateException(
73                  String.format(
74                      "%s: Return type is not void, cannot use @Quietly",
75                      Mnemos.toText(point, true, true)
76                  )
77              );
78          }
79          Object result = null;
80          try {
81              result = point.proceed();
82          // @checkstyle IllegalCatch (1 line)
83          } catch (final Throwable ex) {
84              Logger.warn(
85                  new ImprovedJoinPoint(point).targetize(),
86                  "%[exception]s",
87                  ex
88              );
89          }
90          return result;
91      }
92  }
