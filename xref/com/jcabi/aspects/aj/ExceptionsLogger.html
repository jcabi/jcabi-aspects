30  package com.jcabi.aspects.aj;
31  
32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.log.Logger;
34  import org.aspectj.lang.ProceedingJoinPoint;
35  import org.aspectj.lang.annotation.Around;
36  import org.aspectj.lang.annotation.Aspect;
37  
38  /**
39   * Logs all exceptions thrown out of a method.
40   *
41   * @since 0.1.10
42   * @see com.jcabi.aspects.LogExceptions
43   * @checkstyle IllegalThrows (100 lines)
44   * @checkstyle NonStaticMethodCheck (100 lines)
45   */
46  @Aspect
47  @Immutable
48  public final class ExceptionsLogger {
49  
50      /**
51       * Catch exception and log it.
52       *
53       * <p>Try NOT to change the signature of this method, in order to keep
54       * it backward compatible.
55       *
56       * @param point Joint point
57       * @return The result of call
58       * @throws Throwable If something goes wrong inside
59       */
60      @Around
61          (
62              // @checkstyle StringLiteralsConcatenation (2 lines)
63              "execution(* * (..))"
64              + " && @annotation(com.jcabi.aspects.LogExceptions)"
65          )
66      @SuppressWarnings("PMD.AvoidCatchingThrowable")
67      public Object wrap(final ProceedingJoinPoint point) throws Throwable {
68          try {
69              return point.proceed();
70          // @checkstyle IllegalCatch (1 line)
71          } catch (final Throwable ex) {
72              Logger.warn(
73                  new ImprovedJoinPoint(point).targetize(),
74                  "%[exception]s",
75                  ex
76              );
77              throw ex;
78          }
79      }
80  }
