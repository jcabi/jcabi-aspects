56  /**
57   * Cache method results.
58   *
59   * <p>It is an AspectJ aspect and you are not supposed to use it directly. It
60   * is instantiated by AspectJ runtime framework when your code is annotated
61   * with {@link Cacheable} annotation.
62   *
63   * <p>The class is thread-safe.
64   *
65   * @since 0.8
66   */
67  @Aspect
68  @SuppressWarnings(
69      { "PMD.DoNotUseThreads", "PMD.TooManyMethods", "PMD.GodClass" }
70  )
71  public final class MethodCacher {
72  
73      /**
74       * Calling tunnels.
75       * @checkstyle LineLength (2 lines)
76       */
77      private final transient ConcurrentMap<MethodCacher.Key, MethodCacher.Tunnel> tunnels;
78  
79      /**
80       * Save the keys which need update.
81       */
82      private final transient BlockingQueue<MethodCacher.Key> updatekeys;
83  
84      /**
85       * Service that cleans cache.
86       */
87      @SuppressWarnings("PMD.SingularField")
88      private final transient ScheduledExecutorService cleaner;
89  
90      /**
91       * Service that update cache.
92       */
93      @SuppressWarnings("PMD.SingularField")
94      private final transient ScheduledExecutorService updater;
95  
96      /**
97       * Public ctor.
98       */
99      @SuppressWarnings("PMD.ConstructorOnlyInitializesOrCallOtherConstructors")
100     public MethodCacher() {
101         this.tunnels = new ConcurrentHashMap<>(0);
102         this.updatekeys = new LinkedBlockingQueue<>();
103         this.cleaner = Executors.newSingleThreadScheduledExecutor(
104             new NamedThreads(
105                 "cacheable-clean",
106                 "automated cleaning of expired @Cacheable values"
107             )
108         );
109         this.cleaner.scheduleWithFixedDelay(
110             new VerboseRunnable(
111                 this::clean
112             ),
113             1L, 1L, TimeUnit.SECONDS
114         );
115         this.updater = Executors.newSingleThreadScheduledExecutor(
116             new NamedThreads(
117                 "cacheable-update",
118                 "async update of expired @Cacheable values"
119             )
120         );
121         this.updater.schedule(
122             new VerboseRunnable(
123                 this::update
124             ),
125             0L, TimeUnit.SECONDS
126         );
127     }

129     /**
130      * Call the method or fetch from cache.
131      *
132      * <p>Try NOT to change the signature of this method, in order to keep
133      * it backward compatible.
134      *
135      * @param point Joint point
136      * @return The result of call
137      * @throws Throwable If something goes wrong inside
138      * @checkstyle IllegalThrows (4 lines)
139      */
140     @Around("execution(* *(..)) && @annotation(com.jcabi.aspects.Cacheable)")
141     public Object cache(final ProceedingJoinPoint point) throws Throwable {
142         final MethodCacher.Key key = new MethodCacher.Key(point);
143         MethodCacher.Tunnel tunnel;
144         final Method method = ((MethodSignature) point.getSignature())
145             .getMethod();
146         final Cacheable annot = method.getAnnotation(Cacheable.class);
147         synchronized (this.tunnels) {
148             for (final Class<?> before : annot.before()) {
149                 final boolean flag = (Boolean) before.getMethod("flushBefore")
150                     .invoke(method.getClass());
151                 if (flag) {
152                     this.preflush(point);
153                 }
154             }
155             tunnel = this.tunnels.get(key);
156             if (MethodCacher.isCreateTunnel(tunnel)) {
157                 tunnel = new MethodCacher.Tunnel(
158                     point, key, annot.asyncUpdate()
159                 );
160                 this.tunnels.put(key, tunnel);
161             }
162             if (tunnel.expired() && tunnel.asyncUpdate()) {
163                 this.updatekeys.offer(key);
164             }
165             for (final Class<?> after : annot.after()) {
166                 final boolean flag = (Boolean) after.getMethod("flushAfter")
167                     .invoke(method.getClass());
168                 if (flag) {
169                     this.postflush(point);
170                 }
171             }
172         }
173         return tunnel.through();
174     }
175 
176     /**
177      * Whether create a new Tunnel.
178      * @param tunnel MethodCacher.Tunnel
179      * @return Boolean
180      */
181     private static boolean isCreateTunnel(final MethodCacher.Tunnel tunnel) {
182         return tunnel == null || tunnel.expired() && !tunnel.asyncUpdate();
183     }

185     /**
186      * Flush cache.
187      * @param point Join point
188      * @return Value of the method
189      * @since 0.7.14
190      * @deprecated Since 0.7.17, and preflush() should be used
191      * @throws Throwable If something goes wrong inside
192      * @checkstyle IllegalThrows (4 lines)
193      * @checkstyle MethodsOrderCheck (3 lines)
194      */
195     @Deprecated
196     public Object flush(final ProceedingJoinPoint point) throws Throwable {
197         this.preflush(point);
198         return point.proceed();
199     }

201     /**
202      * Flush cache.
203      *
204      * <p>Try NOT to change the signature of this method, in order to keep
205      * it backward compatible.
206      *
207      * @param point Joint point
208      * @since 0.7.14
209      * @checkstyle MethodsOrderCheck (3 lines)
210      */
211     @Before(
212         // @checkstyle StringLiteralsConcatenation (3 lines)
213         "execution(* *(..))"
214         + " && (@annotation(com.jcabi.aspects.Cacheable.Flush)"
215         + " || @annotation(com.jcabi.aspects.Cacheable.FlushBefore))"
216     )
217     public void preflush(final JoinPoint point) {
218         this.flush(point, "before the call");
219     }
220 
221     /**
222      * Flush cache after method execution.
223      *
224      * <p>Try NOT to change the signature of this method, in order to keep
225      * it backward compatible.
226      *
227      * @param point Joint point
228      * @since 0.7.18
229      * @checkstyle MethodsOrderCheck (3 lines)
230      */
231     @After(
232         // @checkstyle StringLiteralsConcatenation (2 lines)
233         "execution(* *(..))"
234         + " && @annotation(com.jcabi.aspects.Cacheable.FlushAfter)"
235     )
236     public void postflush(final JoinPoint point) {
237         this.flush(point, "after the call");
238     }
239 
240     /**
241      * Flush cache.
242      * @param point Joint point
243      * @param when When it happens
244      * @since 0.7.18
245      */
246     private void flush(final JoinPoint point, final String when) {
247         synchronized (this.tunnels) {
248             for (final MethodCacher.Key key : this.tunnels.keySet()) {
249                 if (!key.sameTarget(point)) {
250                     continue;
251                 }
252                 final MethodCacher.Tunnel removed = this.tunnels.remove(key);
253                 final Method method = ((MethodSignature) point.getSignature())
254                     .getMethod();
255                 if (LogHelper.enabled(
256                     key.getLevel(), method.getDeclaringClass()
257                 )) {
258                     LogHelper.log(
259                         key.getLevel(),
260                         method.getDeclaringClass(),
261                         "%s: %s:%s removed from cache %s",
262                         Mnemos.toText(method, point.getArgs(), true, false),
263                         key,
264                         removed,
265                         when
266                     );
267                 }
268             }
269         }
270     }
271 
272     /**
273      * Clean cache.
274      */
275     private void clean() {
276         synchronized (this.tunnels) {
277             for (final Map.Entry<MethodCacher.Key, MethodCacher.Tunnel> entry
278                 : this.tunnels.entrySet()) {
279                 final MethodCacher.Key key = entry.getKey();
280                 if (entry.getValue().expired()
281                     && !entry.getValue().asyncUpdate()) {
282                     final MethodCacher.Tunnel tunnel = this.tunnels.remove(key);
283                     LogHelper.log(
284                         key.getLevel(),
285                         this,
286                         "%s:%s expired in cache",
287                         key,
288                         tunnel
289                     );
290                 }
291             }
292         }
293     }
294 
295     /**
296      * Update cache.
297      */
298     @SuppressWarnings("PMD.AvoidCatchingThrowable")
299     private void update() {
300         while (true) {
301             try {
302                 final MethodCacher.Key key = this.updatekeys.take();
303                 final MethodCacher.Tunnel tunnel = this.tunnels.get(key);
304                 if (tunnel != null && tunnel.expired()) {
305                     final MethodCacher.Tunnel after = tunnel.copy();
306                     after.through();
307                     this.tunnels.put(key, after);
308                 }
309             } catch (final InterruptedException ex) {
310                 LogHelper.log(
311                     Loggable.ERROR,
312                     this,
313                     "%s:%s",
314                     ex.getMessage(),
315                     ex
316                 );
317             // @checkstyle IllegalCatch (1 line)
318             } catch (final Throwable ex) {
319                 LogHelper.log(
320                     Loggable.ERROR,
321                     this,
322                     "Exception message is %s, Exception is %s",
323                     ex.getMessage(),
324                     ex
325                 );
326             }
327         }
328     }

330     /**
331      * Mutable caching/calling tunnel, it is thread-safe.
332      *
333      * @since 0.8
334      */
335     private static final class Tunnel {
336         /**
337          * Proceeding join point.
338          */
339         private final transient ProceedingJoinPoint point;
340 
341         /**
342          * Key related to this tunnel.
343          */
344         private final transient MethodCacher.Key key;
345 
346         /**
347          * Whether asynchronous update.
348          */
349         private final transient boolean asynchupdate;
350 
351         /**
352          * Was it already executed?
353          */
354         private transient boolean executed;
355 
356         /**
357          * When will it expire (moment in time).
358          */
359         private transient long lifetime;
360 
361         /**
362          * Cached value.
363          */
364         private transient Object cached;
365 
366         /**
367          * Public ctor.
368          * @param pnt ProceedingJoinPoint
369          * @param akey MethodCacher.Key
370          * @param aupdate Boolean
371          */
372         Tunnel(final ProceedingJoinPoint pnt,
373             final MethodCacher.Key akey, final boolean aupdate) {
374             this.point = pnt;
375             this.key = akey;
376             this.asynchupdate = aupdate;
377         }
378 
379         @Override
380         public String toString() {
381             return Mnemos.toText(this.cached, true, false);
382         }
383 
384         /**
385          * Public ctor.
386          * @return MethodCacher.Tunnel
387          */
388         public MethodCacher.Tunnel copy() {
389             return new MethodCacher.Tunnel(
390                 this.point, this.key, this.asynchupdate
391             );
392         }
393 
394         /**
395          * Get a result through the tunnel.
396          * @return The result
397          * @throws Throwable If something goes wrong inside
398          * @checkstyle IllegalThrows (5 lines)
399          */
400         @SuppressWarnings("PMD.AvoidSynchronizedAtMethodLevel")
401         public synchronized Object through() throws Throwable {
402             if (!this.executed) {
403                 final long start = System.currentTimeMillis();
404                 this.cached = this.point.proceed();
405                 final Method method = ((MethodSignature) this.point.getSignature())
406                     .getMethod();
407                 final Cacheable annot = method.getAnnotation(Cacheable.class);
408                 final String suffix;
409                 if (annot.forever()) {
410                     this.lifetime = Long.MAX_VALUE;
411                     suffix = "valid forever";
412                 } else if (annot.lifetime() == 0) {
413                     this.lifetime = 0L;
414                     suffix = "invalid immediately";
415                 } else {
416                     final long msec = annot.unit().toMillis(
417                         (long) annot.lifetime()
418                     );
419                     this.lifetime = start + msec;
420                     suffix = Logger.format("valid for %[ms]s", msec);
421                 }
422                 final Class<?> type = method.getDeclaringClass();
423                 if (LogHelper.enabled(this.key.getLevel(), type)) {
424                     LogHelper.log(
425                         this.key.getLevel(),
426                         type,
427                         "%s: %s cached in %[ms]s, %s",
428                         Mnemos.toText(
429                             method, this.point.getArgs(), true, false
430                         ),
431                         Mnemos.toText(this.cached, true, false),
432                         System.currentTimeMillis() - start,
433                         suffix
434                     );
435                 }
436                 this.executed = true;
437             }
438             return this.key.through(this.cached);
439         }
440 
441         /**
442          * Is it expired already?
443          * @return TRUE if expired
444          */
445         public boolean expired() {
446             return this.executed && this.lifetime < System.currentTimeMillis();
447         }
448 
449         /**
450          * Whether asynchronous update.
451          * @return TRUE if asynchronous update
452          */
453         public boolean asyncUpdate() {
454             return this.asynchupdate;
455         }
456     }

458     /**
459      * Key of a callable target.
460      *
461      * @since 0.8
462      */
463     private static final class Key {
464         /**
465          * When instantiated.
466          */
467         private final transient long start;
468 
469         /**
470          * How many times the key was already accessed.
471          */
472         private final transient AtomicInteger accessed;
473 
474         /**
475          * Method.
476          */
477         private final transient Method method;
478 
479         /**
480          * Object callable (or class, if static method).
481          */
482         private final transient Object target;
483 
484         /**
485          * Arguments.
486          */
487         private final transient Object[] arguments;
488 
489         /**
490          * Log level.
491          */
492         private final int level;
493 
494         /**
495          * Public ctor.
496          * @param point Joint point
497          */
498         @SuppressWarnings("PMD.ConstructorOnlyInitializesOrCallOtherConstructors")
499         Key(final JoinPoint point) {
500             this.start = System.currentTimeMillis();
501             this.accessed = new AtomicInteger();
502             this.method = ((MethodSignature) point.getSignature()).getMethod();
503             this.target = MethodCacher.Key.targetize(point);
504             this.arguments = point.getArgs();
505             if (this.method.isAnnotationPresent(Loggable.class)) {
506                 this.level = this.method.getAnnotation(Loggable.class).value();
507             } else {
508                 this.level = Loggable.DEBUG;
509             }
510         }
511 
512         @Override
513         public String toString() {
514             return Mnemos.toText(this.method, this.arguments, true, false);
515         }
516 
517         @Override
518         public int hashCode() {
519             return this.method.hashCode();
520         }
521 
522         @Override
523         public boolean equals(final Object obj) {
524             final boolean equals;
525             if (this == obj) {
526                 equals = true;
527             } else if (obj instanceof MethodCacher.Key) {
528                 final MethodCacher.Key key = (MethodCacher.Key) obj;
529                 equals = key.method.equals(this.method)
530                     && this.target.equals(key.target)
531                     && Arrays.deepEquals(key.arguments, this.arguments);
532             } else {
533                 equals = false;
534             }
535             return equals;
536         }
537 
538         /**
539          * Send a result through, with necessary logging.
540          * @param result The result to send through
541          * @return The same result/object
542          */
543         public Object through(final Object result) {
544             final int hit = this.accessed.getAndIncrement();
545             final Class<?> type = this.method.getDeclaringClass();
546             if (hit > 0 && LogHelper.enabled(this.level, type)) {
547                 LogHelper.log(
548                     this.level,
549                     type,
550                     "%s: %s from cache (hit #%d, %[ms]s old)",
551                     this,
552                     Mnemos.toText(result, true, false),
553                     hit,
554                     System.currentTimeMillis() - this.start
555                 );
556             }
557             return result;
558         }
559 
560         /**
561          * Is it related to the same target?
562          * @param point Proceeding point
563          * @return True if the target is the same
564          */
565         public boolean sameTarget(final JoinPoint point) {
566             return MethodCacher.Key.targetize(point).equals(this.target);
567         }
568 
569         /**
570          * Calculate its target.
571          * @param point Proceeding point
572          * @return The target
573          */
574         private static Object targetize(final JoinPoint point) {
575             final Object tgt;
576             final Method method = ((MethodSignature) point.getSignature()).getMethod();
577             if (Modifier.isStatic(method.getModifiers())) {
578                 tgt = method.getDeclaringClass();
579             } else {
580                 tgt = point.getTarget();
581             }
582             return tgt;
583         }
584 
585         /**
586          * Get log level.
587          * @return Log level of current method.
588          */
589         private int getLevel() {
590             return this.level;
591         }
592     }
