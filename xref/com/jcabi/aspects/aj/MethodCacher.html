51  /**
52   * Cache method results.
53   *
54   * <p>It is an AspectJ aspect and you are not supposed to use it directly. It
55   * is instantiated by AspectJ runtime framework when your code is annotated
56   * with {@link Cacheable} annotation.
57   *
58   * <p>The class is thread-safe.
59   *
60   * @author Yegor Bugayenko (yegor@tpc2.com)
61   * @version $Id: 58b0bc35f1aa88d471605d43a775b610a5b0d797 $
62   * @since 0.8
63   */
64  @Aspect
65  @SuppressWarnings("PMD.TooManyMethods")
66  public final class MethodCacher {
67  
68      /**
69       * Calling tunnels.
70       * @checkstyle LineLength (2 lines)
71       */
72      private final transient ConcurrentMap<MethodCacher.Key, MethodCacher.Tunnel> tunnels =
73          new ConcurrentHashMap<MethodCacher.Key, MethodCacher.Tunnel>(0);
74  
75      /**
76       * Save the keys of caches which need update.
77       */
78      private final transient BlockingQueue<Key> updatekeys =
79          new LinkedBlockingQueue<MethodCacher.Key>();
80  
81      /**
82       * Public ctor.
83       */
84      public MethodCacher() {
85          new UpdateMethodCacher(this.tunnels, this.updatekeys).start();
86      }
87  
88      /**
89       * Call the method or fetch from cache.
90       *
91       * <p>Try NOT to change the signature of this method, in order to keep
92       * it backward compatible.
93       *
94       * @param point Joint point
95       * @return The result of call
96       * @throws Throwable If something goes wrong inside
97       * @checkstyle IllegalThrows (4 lines)
98       */
99      @Around("execution(* *(..)) && @annotation(com.jcabi.aspects.Cacheable)")
100     public Object cache(final ProceedingJoinPoint point) throws Throwable {
101         final MethodCacher.Key key = new MethodCacher.Key(point);
102         MethodCacher.Tunnel tunnel;
103         final Method method = MethodSignature.class
104             .cast(point.getSignature())
105             .getMethod();
106         final Cacheable annot = method.getAnnotation(Cacheable.class);
107         synchronized (this.tunnels) {
108             for (final Class<?> before : annot.before()) {
109                 final boolean flag = Boolean.class.cast(
110                     before.getMethod("flushBefore").invoke(method.getClass())
111                 );
112                 if (flag) {
113                     this.preflush(point);
114                 }
115             }
116             tunnel = this.tunnels.get(key);
117             if (this.isCreateTunnel(tunnel)) {
118                 tunnel = new MethodCacher.Tunnel(
119                     point, key, annot.asyncUpdate()
120                 );
121                 this.tunnels.put(key, tunnel);
122             }
123             if (tunnel.expired() && tunnel.asyncUpdate()) {
124                 this.updatekeys.offer(key);
125             }
126             for (final Class<?> after : annot.after()) {
127                 final boolean flag = Boolean.class.cast(
128                     after.getMethod("flushAfter").invoke(method.getClass())
129                 );
130                 if (flag) {
131                     this.postflush(point);
132                 }
133             }
134         }
135         return tunnel.through();
136     }

138     /**
139      * Flush cache.
140      * @param point Join point
141      * @return Value of the method
142      * @since 0.7.14
143      * @deprecated Since 0.7.17, and preflush() should be used
144      * @throws Throwable If something goes wrong inside
145      * @checkstyle IllegalThrows (3 lines)
146      */
147     @Deprecated
148     public Object flush(final ProceedingJoinPoint point) throws Throwable {
149         this.preflush(point);
150         return point.proceed();
151     }

153     /**
154      * Flush cache.
155      *
156      * <p>Try NOT to change the signature of this method, in order to keep
157      * it backward compatible.
158      *
159      * @param point Joint point
160      * @since 0.7.14
161      */
162     @Before
163         (
164             // @checkstyle StringLiteralsConcatenation (3 lines)
165             "execution(* *(..))"
166             + " && (@annotation(com.jcabi.aspects.Cacheable.Flush)"
167             + " || @annotation(com.jcabi.aspects.Cacheable.FlushBefore))"
168         )
169     public void preflush(final JoinPoint point) {
170         this.flush(point, "before the call");
171     }
172 
173     /**
174      * Flush cache after method execution.
175      *
176      * <p>Try NOT to change the signature of this method, in order to keep
177      * it backward compatible.
178      *
179      * @param point Joint point
180      * @since 0.7.18
181      */
182     @After
183         (
184             // @checkstyle StringLiteralsConcatenation (2 lines)
185             "execution(* *(..))"
186             + " && @annotation(com.jcabi.aspects.Cacheable.FlushAfter)"
187         )
188     public void postflush(final JoinPoint point) {
189         this.flush(point, "after the call");
190     }
191 
192     /**
193      * Flush cache.
194      * @param point Joint point
195      * @param when When it happens
196      * @since 0.7.18
197      */
198     private void flush(final JoinPoint point, final String when) {
199         synchronized (this.tunnels) {
200             for (final MethodCacher.Key key : this.tunnels.keySet()) {
201                 if (!key.sameTarget(point)) {
202                     continue;
203                 }
204                 final MethodCacher.Tunnel removed = this.tunnels.remove(key);
205                 final Method method = MethodSignature.class
206                     .cast(point.getSignature())
207                     .getMethod();
208                 if (LogHelper.enabled(
209                     key.getLevel(), method.getDeclaringClass()
210                 )) {
211                     LogHelper.log(
212                         key.getLevel(),
213                         method.getDeclaringClass(),
214                         "%s: %s:%s removed from cache %s",
215                         Mnemos.toText(method, point.getArgs(), true, false),
216                         key,
217                         removed,
218                         when
219                     );
220                 }
221             }
222         }
223     }
224 
225     /**
226      * Whether create a new Tunnel.
227      * @param tunnel MethodCacher.Tunnel
228      * @return Boolean
229      */
230     private boolean isCreateTunnel(final MethodCacher.Tunnel tunnel) {
231         return tunnel == null || (tunnel.expired() && !tunnel.asyncUpdate());
232     }

234     /**
235      * Mutable caching/calling tunnel, it is thread-safe.
236      */
237     protected static final class Tunnel {
238         /**
239          * Proceeding join point.
240          */
241         private final transient ProceedingJoinPoint point;
242         /**
243          * Key related to this tunnel.
244          */
245         private final transient MethodCacher.Key key;
246         /**
247          * Whether asynchronous update.
248          */
249         private final transient boolean async;
250         /**
251          * Was it already executed?
252          */
253         private transient boolean executed;
254         /**
255          * When will it expire (moment in time).
256          */
257         private transient long lifetime;
258         /**
259          * Cached value.
260          */
261         private transient Object cached;
262 
263         /**
264          * Public ctor.
265          * @param pnt ProceedingJoinPoint
266          * @param akey MethodCacher.Key
267          * @param asy Boolean
268          */
269         Tunnel(final ProceedingJoinPoint pnt,
270             final MethodCacher.Key akey, final boolean asy) {
271             this.point = pnt;
272             this.key = akey;
273             this.async = asy;
274         }
275 
276         @Override
277         public String toString() {
278             return Mnemos.toText(this.cached, true, false);
279         }
280 
281         /**
282          * Get a new instance.
283          * @return MethodCacher.Tunnel
284          */
285         public Tunnel copy() {
286             return new Tunnel(
287                 this.point, this.key, this.async
288             );
289         }
290         /**
291          * Get a result through the tunnel.
292          * @return The result
293          * @throws Throwable If something goes wrong inside
294          * @checkstyle IllegalThrows (5 lines)
295          */
296         @SuppressWarnings("PMD.AvoidSynchronizedAtMethodLevel")
297         public synchronized Object through() throws Throwable {
298             if (!this.executed) {
299                 final long start = System.currentTimeMillis();
300                 this.cached = this.point.proceed();
301                 final Method method = MethodSignature.class
302                     .cast(this.point.getSignature())
303                     .getMethod();
304                 final Cacheable annot = method.getAnnotation(Cacheable.class);
305                 final String suffix;
306                 if (annot.forever()) {
307                     this.lifetime = Long.MAX_VALUE;
308                     suffix = "valid forever";
309                 } else if (annot.lifetime() == 0) {
310                     this.lifetime = 0L;
311                     suffix = "invalid immediately";
312                 } else {
313                     final long msec = annot.unit().toMillis(
314                         (long) annot.lifetime()
315                     );
316                     this.lifetime = start + msec;
317                     suffix = Logger.format("valid for %[ms]s", msec);
318                 }
319                 final Class<?> type = method.getDeclaringClass();
320                 if (LogHelper.enabled(this.key.getLevel(), type)) {
321                     LogHelper.log(
322                         this.key.getLevel(),
323                         type,
324                         "%s: %s cached in %[ms]s, %s",
325                         Mnemos.toText(
326                             method, this.point.getArgs(), true, false
327                         ),
328                         Mnemos.toText(this.cached, true, false),
329                         System.currentTimeMillis() - start,
330                         suffix
331                     );
332                 }
333                 this.executed = true;
334             }
335             return this.key.through(this.cached);
336         }
337         /**
338          * Is it expired already?
339          * @return TRUE if expired
340          */
341         public boolean expired() {
342             return this.executed && this.lifetime < System.currentTimeMillis();
343         }
344 
345         /**
346          * Whether asynchronous update.
347          * @return TRUE if asynchronous update
348          */
349         public boolean asyncUpdate() {
350             return this.async;
351         }
352     }

354     /**
355      * Key of a callable target.
356      */
357     protected static final class Key {
358         /**
359          * When instantiated.
360          */
361         private final transient long start = System.currentTimeMillis();
362         /**
363          * How many times the key was already accessed.
364          */
365         private final transient AtomicInteger accessed = new AtomicInteger();
366         /**
367          * Method.
368          */
369         private final transient Method method;
370         /**
371          * Object callable (or class, if static method).
372          */
373         private final transient Object target;
374         /**
375          * Arguments.
376          */
377         private final transient Object[] arguments;
378 
379         /**
380          * Log level.
381          */
382         private final int level;
383 
384         /**
385          * Public ctor.
386          * @param point Joint point
387          */
388         Key(final JoinPoint point) {
389             this.method = MethodSignature.class
390                 .cast(point.getSignature()).getMethod();
391             this.target = MethodCacher.Key.targetize(point);
392             this.arguments = point.getArgs();
393             if (this.method.isAnnotationPresent(Loggable.class)) {
394                 this.level = this.method.getAnnotation(Loggable.class).value();
395             } else {
396                 this.level = Loggable.DEBUG;
397             }
398         }
399 
400         /**
401          * Get log level.
402          * @return Log level of current method.
403          */
404         public int getLevel() {
405             return this.level;
406         }
407 
408         @Override
409         public String toString() {
410             return Mnemos.toText(this.method, this.arguments, true, false);
411         }
412         @Override
413         public int hashCode() {
414             return this.method.hashCode();
415         }
416         @Override
417         public boolean equals(final Object obj) {
418             final boolean equals;
419             if (this == obj) {
420                 equals = true;
421             } else if (obj instanceof MethodCacher.Key) {
422                 final MethodCacher.Key key = MethodCacher.Key.class.cast(obj);
423                 equals = key.method.equals(this.method)
424                     && this.target.equals(key.target)
425                     && Arrays.deepEquals(key.arguments, this.arguments);
426             } else {
427                 equals = false;
428             }
429             return equals;
430         }
431         /**
432          * Send a result through, with necessary logging.
433          * @param result The result to send through
434          * @return The same result/object
435          */
436         public Object through(final Object result) {
437             final int hit = this.accessed.getAndIncrement();
438             final Class<?> type = this.method.getDeclaringClass();
439             if (hit > 0 && LogHelper.enabled(this.level, type)) {
440                 LogHelper.log(
441                     this.level,
442                     type,
443                     "%s: %s from cache (hit #%d, %[ms]s old)",
444                     this,
445                     Mnemos.toText(result, true, false),
446                     hit,
447                     System.currentTimeMillis() - this.start
448                 );
449             }
450             return result;
451         }
452         /**
453          * Is it related to the same target?
454          * @param point Proceeding point
455          * @return True if the target is the same
456          */
457         public boolean sameTarget(final JoinPoint point) {
458             return MethodCacher.Key.targetize(point).equals(this.target);
459         }
460         /**
461          * Calculate its target.
462          * @param point Proceeding point
463          * @return The target
464          */
465         private static Object targetize(final JoinPoint point) {
466             final Object tgt;
467             final Method method = MethodSignature.class
468                 .cast(point.getSignature()).getMethod();
469             if (Modifier.isStatic(method.getModifiers())) {
470                 tgt = method.getDeclaringClass();
471             } else {
472                 tgt = point.getTarget();
473             }
474             return tgt;
475         }
