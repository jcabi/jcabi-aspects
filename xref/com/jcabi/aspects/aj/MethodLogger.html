48  /**
49   * Logs method calls.
50   *
51   * <p>It is an AspectJ aspect and you are not supposed to use it directly. It
52   * is instantiated by AspectJ runtime framework when your code is annotated
53   * with {@link Loggable} annotation.
54   *
55   * @since 0.7.2
56   * @checkstyle IllegalThrows (500 lines)
57   */
58  @Aspect
59  @SuppressWarnings
60      (
61          {
62              "PMD.AvoidCatchingThrowable",
63              "PMD.TooManyMethods",
64              "PMD.CyclomaticComplexity"
65          }
66      )
67  public final class MethodLogger {
68  
69      /**
70       * Currently running methods.
71       */
72      private final transient Set<MethodLogger.Marker> running;
73  
74      /**
75       * Public ctor.
76       */
77      @SuppressWarnings("PMD.DoNotUseThreads")
78      public MethodLogger() {
79          this.running = new ConcurrentSkipListSet<>();
80          final ScheduledExecutorService monitor =
81              Executors.newSingleThreadScheduledExecutor(
82                  new NamedThreads(
83                      "loggable",
84                      "watching of @Loggable annotated methods"
85                  )
86              );
87          monitor.scheduleWithFixedDelay(
88              new FutureTask<Void>(
89                  new VerboseRunnable(
90                      () -> {
91                          for (final MethodLogger.Marker marker
92                              : this.running) {
93                              marker.monitor();
94                          }
95                      }
96                  ), null
97              ) {
98                  @Override
99                  protected void done() {
100                     Logger.debug(this, "Logging monitor thread interrupted");
101                     monitor.shutdown();
102                 }
103             },
104             1L, 1L, TimeUnit.SECONDS
105         );
106     }

108     /**
109      * Log methods in a class.
110      *
111      * <p>Try NOT to change the signature of this method, in order to keep
112      * it backward compatible.
113      *
114      * @param point Joint point
115      * @return The result of call
116      * @throws Throwable If something goes wrong inside
117      */
118     @Around
119         (
120             // @checkstyle StringLiteralsConcatenation (7 lines)
121             "execution(public * (@com.jcabi.aspects.Loggable *).*(..))"
122             + " && !execution(String *.toString())"
123             + " && !execution(int *.hashCode())"
124             + " && !execution(boolean *.canEqual(Object))"
125             + " && !execution(boolean *.equals(Object))"
126             + " && !cflow(call(com.jcabi.aspects.aj.MethodLogger.new()))"
127         )
128     public Object wrapClass(final ProceedingJoinPoint point) throws Throwable {
129         final Method method =
130             ((MethodSignature) point.getSignature()).getMethod();
131         final Object output;
132         if (method.isAnnotationPresent(Loggable.class)) {
133             output = point.proceed();
134         } else {
135             output = this.wrap(
136                 point,
137                 method,
138                 method.getDeclaringClass().getAnnotation(Loggable.class)
139             );
140         }
141         return output;
142     }
143 
144     /**
145      * Log individual methods.
146      *
147      * <p>Try NOT to change the signature of this method, in order to keep
148      * it backward compatible.
149      *
150      * @param point Joint point
151      * @return The result of call
152      * @throws Throwable If something goes wrong inside
153      */
154     @Around
155         (
156             // @checkstyle StringLiteralsConcatenation (2 lines)
157             "(execution(* *(..)) || initialization(*.new(..)))"
158             + " && @annotation(com.jcabi.aspects.Loggable)"
159         )
160     @SuppressWarnings("PMD.AvoidCatchingThrowable")
161     public Object wrapMethod(final ProceedingJoinPoint point) throws Throwable {
162         final Method method =
163             ((MethodSignature) point.getSignature()).getMethod();
164         return this.wrap(point, method, method.getAnnotation(Loggable.class));
165     }

167     /**
168      * Catch exception and re-call the method.
169      * @param point Joint point
170      * @param method The method
171      * @param annotation The annotation
172      * @return The result of call
173      * @throws Throwable If something goes wrong inside
174      * @checkstyle ExecutableStatementCountCheck (100 lines)
175      */
176     private Object wrap(final ProceedingJoinPoint point, final Method method,
177         final Loggable annotation) throws Throwable {
178         if (Thread.interrupted()) {
179             throw new IllegalStateException(
180                 String.format(
181                     "thread '%s' in group '%s' interrupted",
182                     Thread.currentThread().getName(),
183                     Thread.currentThread().getThreadGroup().getName()
184                 )
185             );
186         }
187         final long start = System.nanoTime();
188         final MethodLogger.Marker marker =
189             new MethodLogger.Marker(point, annotation);
190         this.running.add(marker);
191         int level = annotation.value();
192         try {
193             final Object logger = MethodLogger.logger(method, annotation.name());
194             if (annotation.prepend()) {
195                 LogHelper.log(
196                     level,
197                     logger,
198                     new StringBuilder(
199                         Mnemos.toText(
200                             point,
201                             annotation.trim(),
202                             annotation.skipArgs(),
203                             annotation.logThis()
204                         )
205                     ).append(": entered").toString()
206                 );
207             }
208             final Object result = point.proceed();
209             final long nano = System.nanoTime() - start;
210             if (LogHelper.enabled(level, logger)
211                 || MethodLogger.over(annotation, nano)) {
212                 if (MethodLogger.over(annotation, nano)) {
213                     level = Loggable.WARN;
214                 }
215                 LogHelper.log(
216                     level, logger,
217                     MethodLogger.message(point, method, annotation, result, nano)
218                 );
219             }
220             return result;
221         // @checkstyle IllegalCatch (1 line)
222         } catch (final Throwable ex) {
223             if (!MethodLogger.contains(annotation.ignore(), ex)
224                 && !ex.getClass().isAnnotationPresent(Loggable.Quiet.class)) {
225                 final StackTraceElement[] traces = ex.getStackTrace();
226                 final String origin;
227                 if (traces.length > 0) {
228                     final StackTraceElement trace = traces[0];
229                     origin = MethodLogger.oneText(trace);
230                 } else {
231                     origin = "somewhere";
232                 }
233                 LogHelper.log(
234                     level,
235                     method.getDeclaringClass(),
236                     Logger.format(
237                         "%s: thrown %s out of %s in %[nano]s",
238                         Mnemos.toText(
239                             point,
240                             annotation.trim(),
241                             annotation.skipArgs(),
242                             annotation.logThis()
243                         ),
244                         Mnemos.toText(ex),
245                         origin,
246                         System.nanoTime() - start
247                     )
248                 );
249             }
250             throw ex;
251         } finally {
252             this.running.remove(marker);
253         }
254     }

256     /**
257      * Has time for method execution passed.
258      * @param annotation Loggable annotation.
259      * @param nano Execution time.
260      * @return Is over time limit.
261      */
262     private static boolean over(final Loggable annotation, final long nano) {
263         return nano > annotation.unit().toNanos(
264             (long) annotation.limit()
265         );
266     }

268     /**
269      * Prepared message for log.
270      * @param point JointPoint to use.
271      * @param method Method for which to log.
272      * @param annotation Loggable annotation.
273      * @param result Method result.
274      * @param nano Method execution time.
275      * @return Log message.
276      * @checkstyle ParameterNumberCheck (3 lines)
277      */
278     private static String message(final ProceedingJoinPoint point, final Method method,
279         final Loggable annotation, final Object result, final long nano) {
280         final StringBuilder msg = new StringBuilder(0);
281         msg.append(
282             Mnemos.toText(
283                 point,
284                 annotation.trim(),
285                 annotation.skipArgs(),
286                 annotation.logThis()
287             )
288         ).append(':');
289         if (!method.getReturnType().equals(Void.TYPE)) {
290             msg.append(' ').append(
291                 Mnemos.toText(
292                     result,
293                     annotation.trim(),
294                     annotation.skipResult()
295                 )
296             );
297         }
298         msg.append(
299             Logger.format(
300                 String.format(
301                     " in %%[nano].%ds", annotation.precision()
302                 ),
303                 nano
304             )
305         );
306         if (MethodLogger.over(annotation, nano)) {
307             msg.append(" (too slow!)");
308         }
309         return msg.toString();
310     }

312     /**
313      * Get the destination logger for this method.
314      * @param method The method
315      * @param name The Loggable annotation
316      * @return The logger that will be used
317      */
318     private static Object logger(final Method method, final CharSequence name) {
319         final Object source;
320         if (name.length() == 0) {
321             source = method.getDeclaringClass();
322         } else {
323             source = name;
324         }
325         return source;
326     }

328     /**
329      * Checks whether array of types contains given type.
330      * @param array Array of them
331      * @param exp The exception to find
332      * @return TRUE if it's there
333      */
334     private static boolean contains(final Class<? extends Throwable>[] array,
335         final Throwable exp) {
336         boolean contains = false;
337         for (final Class<? extends Throwable> type : array) {
338             if (MethodLogger.instanceOf(exp.getClass(), type)) {
339                 contains = true;
340                 break;
341             }
342         }
343         return contains;
344     }
345 
346     /**
347      * The type is an instance of another type?
348      * @param child The child type
349      * @param parent Parent type
350      * @return TRUE if child is really a child of a parent
351      */
352     private static boolean instanceOf(final Class<?> child,
353         final Class<?> parent) {
354         boolean instance = child.equals(parent)
355             || child.getSuperclass() != null
356             && MethodLogger.instanceOf(child.getSuperclass(), parent);
357         if (!instance) {
358             for (final Class<?> iface : child.getInterfaces()) {
359                 instance = MethodLogger.instanceOf(iface, parent);
360                 if (instance) {
361                     break;
362                 }
363             }
364         }
365         return instance;
366     }

368     /**
369      * Textualize a stacktrace.
370      * @param trace Array of stacktrace elements
371      * @return The text
372      */
373     private static String allText(final StackTraceElement... trace) {
374         final StringBuilder text = new StringBuilder(0);
375         for (final StackTraceElement element : trace) {
376             if (text.length() > 0) {
377                 text.append(", ");
378             }
379             text.append(MethodLogger.oneText(element));
380         }
381         return text.toString();
382     }
383 
384     /**
385      * Textualize a stacktrace.
386      * @param trace One stacktrace element
387      * @return The text
388      */
389     private static String oneText(final StackTraceElement trace) {
390         return String.format(
391             "%s#%s[%d]",
392             trace.getClassName(),
393             trace.getMethodName(),
394             trace.getLineNumber()
395         );
396     }

398     /**
399      * Marker of a running method.
400      * @since 0.0.0
401      */
402     private static final class Marker
403         implements Comparable<MethodLogger.Marker> {
404 
405         /**
406          * When the method was started, in milliseconds.
407          */
408         private final transient long started;
409 
410         /**
411          * Which monitoring cycle was logged recently.
412          */
413         private final transient AtomicInteger logged;
414 
415         /**
416          * The thread it's running in.
417          */
418         @SuppressWarnings("PMD.DoNotUseThreads")
419         private final transient Thread thread;
420 
421         /**
422          * Joint point.
423          */
424         private final transient ProceedingJoinPoint point;
425 
426         /**
427          * Annotation.
428          */
429         private final transient Loggable annotation;
430 
431         /**
432          * Public ctor.
433          * @param pnt Joint point
434          * @param annt Annotation
435          */
436         protected Marker(final ProceedingJoinPoint pnt, final Loggable annt) {
437             this.started = System.currentTimeMillis();
438             this.logged = new AtomicInteger();
439             this.point = pnt;
440             this.annotation = annt;
441             this.thread = Thread.currentThread();
442         }
443 
444         /**
445          * Monitor it's status and log the problem, if any.
446          */
447         public void monitor() {
448             final TimeUnit unit = this.annotation.unit();
449             final long threshold = this.annotation.limit();
450             final long age = unit.convert(
451                 System.currentTimeMillis() - this.started, TimeUnit.MILLISECONDS
452             );
453             final int cycle = (int) ((age - threshold) / threshold);
454             if (cycle > this.logged.get()) {
455                 final Method method = ((MethodSignature) this.point.getSignature()).getMethod();
456                 Logger.warn(
457                     method.getDeclaringClass(),
458                     "%s: takes more than %[ms]s, %[ms]s already, thread=%s/%s",
459                     Mnemos.toText(this.point, true, this.annotation.skipArgs()),
460                     TimeUnit.MILLISECONDS.convert(threshold, unit),
461                     TimeUnit.MILLISECONDS.convert(age, unit),
462                     this.thread.getName(),
463                     this.thread.getState()
464                 );
465                 Logger.debug(
466                     method.getDeclaringClass(),
467                     "%s: thread %s/%s stacktrace: %s",
468                     Mnemos.toText(this.point, true, this.annotation.skipArgs()),
469                     this.thread.getName(),
470                     this.thread.getState(),
471                     MethodLogger.allText(this.thread.getStackTrace())
472                 );
473                 this.logged.set(cycle);
474             }
475         }
476 
477         @Override
478         public int hashCode() {
479             return this.point.hashCode();
480         }
481 
482         @Override
483         public boolean equals(final Object obj) {
484             return obj == this || ((MethodLogger.Marker) obj)
485                 .point.equals(this.point);
486         }
487 
488         @Override
489         public int compareTo(final MethodLogger.Marker marker) {
490             int cmp = 0;
491             if (this.started < marker.started) {
492                 cmp = 1;
493             } else if (this.started > marker.started) {
494                 cmp = -1;
495             }
496             return cmp;
497         }
498     }
