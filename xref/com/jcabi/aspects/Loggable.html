30  package com.jcabi.aspects;
31  
32  import java.lang.annotation.Documented;
33  import java.lang.annotation.ElementType;
34  import java.lang.annotation.Retention;
35  import java.lang.annotation.RetentionPolicy;
36  import java.lang.annotation.Target;
37  import java.util.concurrent.TimeUnit;
38  
39  /**
40   * Makes a method loggable via {@link com.jcabi.log.Logger}.
41   *
42   * <p>For example, this {@code load()} method produces a log line
43   * on every call:
44   *
45   * <pre> &#64;Loggable
46   * String load(String resource) throws IOException {
47   *   return "something";
48   * }</pre>
49   *
50   * <p>You can configure the level of logging:
51   *
52   * <pre> &#64;Loggable(Loggable.DEBUG)
53   * void save(String resource) throws IOException {
54   *   // do something
55   * }</pre>
56   *
57   * <p>Since version 0.7.6, you can specify a maximum execution time limit for
58   * a method. If such a limit is reached a logging message will be issued with
59   * a {@code WARN} priority. It is a very convenient mechanism for profiling
60   * applications in production. Default value of a limit is 1 second.
61   *
62   * <pre> &#64;Loggable(limit = 2)
63   * void save(String resource) throws IOException {
64   *   // do something, potentially slow
65   * }</pre>
66   *
67   * <p>Since version 0.7.14 you can change the time unit for the "limit"
68   * parameter. Default unit of measurement is a second:
69   *
70   * <pre> &#64;Loggable(limit = 200, unit = TimeUnit.MILLISECONDS)
71   * void save(String resource) throws IOException {
72   *   // do something, potentially slow
73   * }</pre>
74   *
75   * <p>Since version 0.7.17 you can ignore certain exception types, and they
76   * won't be logged when thrown. It is very useful when exceptions are used
77   * to control flow (which is not a good practice, but is still used in
78   * some frameworks, for example in JAX-RS):
79   *
80   * <pre> &#64;Loggable(ignore = WebApplicationException.class)
81   * String get() {
82   *   if (not_logged_in()) {
83   *     throw new WebApplicationException(forward_to_login_page());
84   *   }
85   * }</pre>
86   *
87   * <p>Since version 0.8 you can mark some exceptions as "always to be ignored",
88   * using {@link Loggable.Quiet} annotation.
89   *
90   * @since 0.7.2
91   * @see com.jcabi.log.Logger
92   * @see <a href="http://aspects.jcabi.com">http://aspects.jcabi.com/</a>
93   * @see <a href="http://www.yegor256.com/2014/06/01/aop-aspectj-java-method-logging.html">Java Method Logging with AOP and Annotations, by Yegor Bugayenko</a>
94   */
95  @Documented
96  @Retention(RetentionPolicy.RUNTIME)
97  @Target({ ElementType.METHOD, ElementType.TYPE })
98  @SuppressWarnings({
99      "PMD.VariableNamingConventions", "PMD.RedundantFieldInitializer"
100 })
101 public @interface Loggable {
102 
103     /**
104      * TRACE level of logging.
105      */
106     int TRACE = 0;
107 
108     /**
109      * DEBUG level of logging.
110      */
111     int DEBUG = 1;
112 
113     /**
114      * INFO level of logging.
115      */
116     int INFO = 2;
117 
118     /**
119      * WARN level of logging.
120      */
121     int WARN = 3;
122 
123     /**
124      * ERROR level of logging.
125      */
126     int ERROR = 4;
127 
128     /**
129      * Level of logging.
130      *
131      * @return The log level
132      */
133     int value() default Loggable.INFO;
134 
135     /**
136      * Maximum amount allowed for this method (a warning will be
137      * issued if it takes longer).
138      * @since 0.7.6
139      * @return The limit
140      */
141     int limit() default 1;
142 
143     /**
144      * Time unit for the limit.
145      * @since 0.7.14
146      * @return The time unit
147      */
148     TimeUnit unit() default TimeUnit.MINUTES;
149 
150     /**
151      * Shall we trim long texts in order to make log lines more readable?
152      * @since 0.7.13
153      * @return The flag
154      */
155     boolean trim() default true;
156 
157     /**
158      * Method entry moment should be reported as well (by default only
159      * an exit moment is reported).
160      * @since 0.7.16
161      * @return The flag
162      */
163     boolean prepend() default false;
164 
165     /**
166      * List of exception types, which should not be logged if thrown.
167      *
168      * <p>You can also mark some exception types as "always to be ignored",
169      * using {@link Loggable.Quiet} annotation.
170      *
171      * @since 0.7.17
172      * @return Array of types
173      */
174     Class<? extends Throwable>[] ignore() default { };
175 
176     /**
177      * Skip logging of result, replacing it with dots?
178      * @since 0.7.19
179      * @return The flag
180      */
181     boolean skipResult() default false;
182 
183     /**
184      * Skip logging of arguments, replacing them all with dots?
185      * @since 0.7.19
186      * @return The flag
187      */
188     boolean skipArgs() default false;
189 
190     /**
191      * Add toString() result to log line.
192      * @since 0.8.1
193      * @return The flag
194      */
195     boolean logThis() default false;
196 
197     /**
198      * The precision (number of fractional digits) to be used when displaying
199      * the measured execution time.
200      * @since 0.18
201      * @return The precision
202      */
203     int precision() default 2;
204 
205     /**
206      * The name of the logger to be used. If not specified, defaults to the
207      * class name of the annotated class or method.
208      * @since 0.18
209      * @return The logger's name
210      */
211     String name() default "";
212 
213     /**
214      * Identifies an exception that is never logged by {@link Loggable} if/when
215      * being thrown out of an annotated method.
216      *
217      * <p>Sometimes exceptions are used as flow control instruments (although
218      * this may be considered as a bad practice in most casts). In such
219      * situations we don't want to flood log console with error messages. One
220      * of the options is to use {@link Loggable#ignore()} attribute to list
221      * all exception types that should be ignored. However, this
222      * {@link Loggable.Quiet} annotation is more convenient when we want to
223      * ignore one specific exception type in all situations.
224      *
225      * @since 0.8
226      */
227     @Documented
228     @Retention(RetentionPolicy.RUNTIME)
229     @Target(ElementType.TYPE)
230     @interface Quiet {
231     }
232 
233 }
